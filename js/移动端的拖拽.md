  let offsetX = 0,
        offsetY = 0;
      let isDrag = false;
      target.addEventListener(
        'touchstart',
        function(e) {          
          offsetX = e.touches[0].pageX - target.offsetLeft;
          offsetY = e.touches[0].pageY - target.offsetTop;
          isDrag = true;
        },
        // false
        { passive: false }
      );
      target.addEventListener(
        'touchmove',
        function(e) {
          let targetTouch = e.touches[0];
          let mouseX = targetTouch.pageX; //触发点当前的位置
          let mouseY = targetTouch.pageY;

          let moveX = 0; //拖动元素的新位置
          let moveY = 0;

          if (isDrag) {
            moveX = mouseX - offsetX;
            moveY = mouseY - offsetY;

            const targetWidth = target.offsetWidth;
            const targetHeight = target.offsetHeight;

            const maxX = clientWidth - targetWidth;

            const maxY = clientHeight - targetHeight;

            moveX = Math.min(Math.max(0, moveX), maxX);//设定限制区域
            moveY = Math.min(Math.max(0, moveY), maxY);//设定限制区域
            target.style.left = moveX + 'px';
            target.style.top = moveY + 'px';
          }
        },
        // false
        { passive: false }
      );
      target.addEventListener(
        'touchend',
        function(e) {
          isDrag = false;
        },
        // false
        { passive: false }
      );

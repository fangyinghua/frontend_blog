### 数组
1. [989. 数组形式的整数加法](https://leetcode-cn.com/problems/add-to-array-form-of-integer/)

```js
/**
 * @param {number[]} num
 * @param {number} k
 * @return {number[]}
 */
var addToArrayForm = function(num, k) {
   num=num.join('');
   let sum=Number(num)+k;
   const result=[];
   while(sum){
        const n=sum % 10;
        result.push(n);
        sum=parseInt(sum /10);
   }
   let reLen=result.length;
   if(reLen===0){
       result.push(sum);
   }
   return reLen===1?result:result.reverse();
};
```
